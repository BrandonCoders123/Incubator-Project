## Step 3: Add Fetch Items useEffect (around line 2890)

Add this useEffect hook to fetch shop items when entering the shop:

```typescript
// 👇 ADD this useEffect to fetch shop items
useEffect(() => {
  if (gameState.gamePhase !== "shop") return;

  setShopLoading(true);
  setShopError(null);

  fetch("/getItems.php")
    .then((res) => {
      if (!res.ok) throw new Error("Failed to load shop");
      return res.json();
    })
    .then((data: ShopItem[]) => {
      setShopItems(data);
    })
    .catch((err) => {
      console.error(err);
      setShopError("Could not load shop items. Please try again.");
    })
    .finally(() => {
      setShopLoading(false);
    });
}, [gameState.gamePhase]);
```

## Step 4: Add handleBuyItem Function (around line 2920)

Add this function to handle item purchases:

```typescript
// 👇 ADD handleBuyItem function
const handleBuyItem = (item: ShopItem) => {
  // Check currency on the client
  if (gameState.user.currency < item.price) {
    alert("Not enough currency!");
    return;
  }

  // Update game state: deduct currency and add to inventory/cosmetics
  setGameState((prev) => ({
    ...prev,
    user: {
      ...prev.user,
      currency: prev.user.currency - item.price,
      cosmetics: [...prev.user.cosmetics, item.name], // or use item.id
    },
    inventory: [...prev.inventory, item.name],
  }));
```
